class Solution {
  public long solution(long n) {
      long answer = 0;
      String s1 = Long.toString(n);
      String s2 = "";
      int arr[] = new int[s1.length()];
      int temp = 0;
      
      for(int i=0;i<s1.length();i++) {
          arr[i] = s1.charAt(i) - '0';
      }
      
      for(int i=0;i<s1.length()-1;i++) {
          for(int j=0;j<s1.length()-1;j++) {
              if(arr[j+1]>=arr[j]) {
                  temp = arr[j];
                  arr[j] = arr[j+1];
                  arr[j+1] = temp;
              }
          }
      }
      
      for(int i=0;i<s1.length();i++) {
          s2 += arr[i];
      }
      answer = Long.parseLong(s2);
      return answer;
  }
}
